1         '''
2         This module implements histogram equalization of grey-scale images.
3         Images are represented as rectangular lists of lists of pixel intensities.
4         Pixel intensities are integers in the range [0,255] inclusive.
5         
6         Histogram equalization is a technique for increasing contrast in digital
7         images which tends to make details more apparent to the human eye. It works
8         by spreading out the distribution of pixel intensities, producing a (near)
9         linear cumulative distribution.
10        
11        The module provides four main functions which test images for validity; 
12        compute histograms of pixel intensities; compute cumulative distributions
13        of pixel intensities; and perform histogram equalization on images.
14        '''

46        def validate_image(image):
47            '''validate_image(image) -> bool
48        
49            Test if the input image is valid according to the following rules:
50                - An image is a list of lists of integers only (no other types are
51                  valid). 
52                - All rows in the image are of the same length (the image is
53                  rectangular).
54                - All integers i in the image lie in the range 0 <= i <= 255. 
55        
56            Examples:
57        
58            >>> validate_image([]) # the empty image is valid
59            True
60            >>> validate_image([1]) # not a list of lists of integers
61            False
62            >>> validate_image(None) # not a list at all
63            False
64            >>> validate_image([[0,23,97,45], [125,3,12,1], [8,8,8,8]])
65            True
66            >>> validate_image([[1,2,3], [4,5]]) # rows different length
67            False
68            >>> validate_image([[1,2,3], [4,5,256]]) # 256 > maximum intensity
69            False
70            '''
71            # An image must be a list.
72            if type(image) == list:
73                len_first_row = None
74                for row in image:
75                    # Each row in the image must be a list.
76                    if type(row) == list:
77                        # All rows in the image must be of the same length.
78                        if len_first_row == None:
79                            # This is the first row, record its length.
80                            len_first_row = len(row)
81                        # This is not the first row, compare its length
82                        # with the length of the first row.
83                        elif len(row) != len_first_row:
84                            return False
85                        for pixel in row:
86                            # Each pixel in a row must be an integer.
87                            if type(pixel) == int:
88                                # Each integer must be within the allowed intensity
89                                # range.
90                                if pixel < intensity_lower_bound or \
91                                   pixel > intensity_upper_bound:
92                                    return False
93                            else:
94                                return False
95                    else:
96                        return False
97                # The image passed all the tests.
98                return True
99            else:
100               return False

102       def histogram(image):
103           '''histogram(image) -> histogram of pixel intensities
104       
105           Compute a histogram of pixel intensities for the input image.
106       
107           An image is a list of lists of pixel intensities and is assumed to be
108           valid according to the validate_image function (histogram does not
109           check the input image for validity).
110       
111           The output is a histogram of pixel intensities, which is represented
112           as a list of 256 non-negative integers, such that the value at position N
113           counts the number of times a pixel of intensity N appears in the image.
114           If there are no pixels in the image of a particular intensity, then
115           its count will be zero. 
116       
117           Examples:
118       
119           >>> histogram([]) == [0] * 256
120           True
121           >>> histogram([[2, 1, 4], [5, 4, 4]])
122           [0, 1, 1, 0, 3, 1, 0, ... # ending in 249 consecutive zeros.
123           '''
124           # Initialise all the frequency counters to zero, one per pixel intensity.
125           result = (intensity_upper_bound + 1) * [0]
126           # Iterate over all the pixels in the image and count the
127           # frequency of their intensities.
128           for row in image:
129               for pixel in row:
130                   result[pixel] += 1
131           return result

133       def cumulative(histogram):
134           '''cumulative(histogram) -> cumulative distribution
135       
136           Compute the cumulative distribution for the input histogram.
137       
138           The input histogram is (assumed to be) a list of non-negative integers,
139           such that the value at index i represents the frequency of the value i.
140       
141           The output list has the same length as the input list. The value of the
142           output list at index N is equal to the sum of all the values in the input
143           list from index 0 to index N inclusive.
144       
145           Examples:
146       
147           >>> cumulative([])
148           []
149           >>> cumulative([4])
150           [4]
151           >>> cumulative([4, 7, 0, 12])
152           [4, 11, 11, 23]
153           '''
154           # Compute a running sum of the frequencies in the histogram, and store
155           # each value in the result.
156           result = []
157           sum = 0
158           for frequency in histogram:
159               sum += frequency 
160               result.append(sum)
161           return result

163       def equalize(image):
164           '''equalize(image) -> image
165       
166           Compute a histogram-equalized version of the input image.
167       
168           An image is a list of lists of pixel intensities and is assumed to be
169           valid according to the validate_image function (equalize does not
170           check the input image for validity).
171       
172           Let d be the cumulative distribution of the input image, p be the total
173           number of pixels in the image, and d_min be the minimum non-zero value
174           in d. For each pixel in the input image with intensity i, the
175           corresponding pixel in the output image is computed by:
176       
177               round ((d[i] - d_min) / (p - d_min) * 255)
178       
179           If the input image is empty, or all the pixels have the same intensity,
180           equalization would be ineffective so we return the input image
181           unchanged.
182       
183           Examples:
184           
185           >>> equalize([])               # empty image
186           [] 
187           >>> equalize([[10, 10, 10]])   # all pixels the same intensity
188           [[10, 10, 10]] 
189           >>> equalize([[100,101,102]])  # low contrast becomes high contrast
190           [[0, 128, 255]] 
191           '''
192           if len(image) == 0:
193               # The image is empty, return it unchanged.
194               return image
195           cumul_dist = cumulative(histogram(image))
196           # Find the frequency of the minimum pixel intensity in the image.
197           # This is equivalent to the minimum non-zero value in the
198           # cumulative distribution. We take advantage of the fact that,
199           # by definition, the cumulative distribution is monotonically
200           # increasing.
201           freq_min_pixel = 0
202           for frequency in cumul_dist:
203               if frequency != 0:
204                   freq_min_pixel = frequency
205                   break
206           # If the image is not empty the frequency of the minimum 
207           # pixel intensity must be greater than zero.
208           assert(freq_min_pixel > 0)
209           num_pixels = get_width(image) * get_height(image)
210           # If all the pixels are the same intensity then the frequency of
211           # the minimum pixel intensity will equal the total number of
212           # pixels in the image. There is no point trying to equalize
213           # such an image to we return it unchanged.
214           if freq_min_pixel == num_pixels:
215               return image
216           # The difference between the number of pixels in the image and
217           # the frequency of the minimum pixel intensity gives us a
218           # measure of the range of pixel intensities used by original
219           # image relative to the image size.
220           cumul_range = float(num_pixels - freq_min_pixel)
221           # Iterate over the original image and build a new image
222           # by mapping the original pixel intensities to their new
223           # equalized values.
224           result = []
225           for row in image:
226               new_row = []
227               for pixel in row:
228                   # Compute the proportion of total pixel intensities
229                   # which are used up-to-and-including this pixel
230                   # intensity. This yields a number in the range [0,1].
231                   norm_cumul_dist = float(cumul_dist[pixel] - freq_min_pixel)
232                   # Cumul_range is not zero because we have checked
233                   # that freq_min_pixel != num_pixels. This is floating
234                   # point division.
235                   proportion = norm_cumul_dist / cumul_range
236                   # Re-scale the pixel to the full intensity range.
237                   new_pixel = proportion * intensity_upper_bound
238                   new_row.append(int(round(new_pixel)))
239               result.append(new_row)
240           return result
